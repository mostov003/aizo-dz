### الخطوة 2️⃣ - التحقق من CORS
**ملف:** `server.js`  
**الحالة:** ✅ مكون بشكل صحيح

```javascript
// ✅ موجود في server.js:
app.use(cors({
  origin: (origin, callback) => {
    // Allow all origins in development
    const isAllowed = config.allowedOrigins.includes('*') || ...
    if (isAllowed) {
      callback(null, true);
    } else {
      callback(new Error(`Not allowed by CORS`));
    }
  },
  credentials: true
}));
```
